#Write a Qiskit script that creates ∣Φ+⟩, measures both qubits, runs 1024 shots on Aer simulator, and prints counts. (Use modern Qiskit API):

from qiskit_aer import AerSimulator
from qiskit import transpile

qc = QuantumCircuit(2, 2)
qc.h(0)
qc.cx(0, 1)

qc.measure([0, 1], [0, 1])

sim = AerSimulator()
compiled = transpile(qc, sim)
result = sim.run(compiled, shots=1024).result()

counts = result.get_counts()
print(counts)
